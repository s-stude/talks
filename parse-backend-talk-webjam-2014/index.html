Search

Parse.Query

var Box = Parse.Object.extend("Box");
var query = new Parse.Query(Box);
query.equalTo("delivered", false);
query.find({
    success: function(results) { }
});

Parse.Query (cont..)

  • query.equalTo query.notEqualTo
  • query.greaterThan query.lessThan
  • query.lessThanOrEqualTo query.greaterThanOrEqualTo
  • query.limit(10);
  • query.skip(10);
  • query.ascending("price");
  • query.descending("price");
